Usage


                const Grid = window.turbogrid.Grid;
                const container = document.querySelector(".container");
                container.style.width = "100%";
                container.style.height = "120px";
                const grid = new Grid(container);
                grid.setOption({
                    sortField: "name"
                });
                grid.setData({
                    columns: [{
                        id: "name",
                        name: "Name"
                    }, {
                        id: "value",
                        name: "Value"
                    }],
                    rows: [{
                        name: "Row 1",
                        value: "1"
                    }, {
                        name: "Row 2",
                        value: "2"
                    }, {
                        name: 'Row 3',
                        value: '3'
                    }]
                });
                grid.render();
