Install


                npm i turbogrid

Usage


                const container = document.querySelector(".container");
                container.style.width = "500px";
                container.style.height = "100px";
                const grid = new Grid(container);
                grid.setOption({
                    sortField: "name"
                });
                grid.setData({
                    columns: [{
                        id: "name",
                        name: "Name"
                    }, {
                        id: "value",
                        name: "Value"
                    }],
                    rows: [{
                        name: "Row 1",
                        value: "1"
                    }, {
                        name: "Row 2",
                        value: "2"
                    }]
                });
                grid.render();
